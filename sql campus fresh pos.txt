-- Tabel Kasir
CREATE TABLE kasir (
    id_kasir VARCHAR(10) PRIMARY KEY,
    nama VARCHAR(100) NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL -- Sebaiknya di-hash (misal menggunakan bcrypt) di sisi backend
);

-- Tabel Produk
CREATE TABLE produk (
    id_produk VARCHAR(10) PRIMARY KEY,
    nama_produk VARCHAR(150) NOT NULL,
    gambar_produk TEXT,
    kategori VARCHAR(50),
    stock INT DEFAULT 0,
    harga_modal DECIMAL(10, 2) NOT NULL,
    harga_jual DECIMAL(10, 2) NOT NULL,
    status_aktif BOOLEAN DEFAULT TRUE,
    kontak_penjual VARCHAR(15)
);

-- Tabel Pesanan (Data Tagihan & Header Transaksi)
CREATE TABLE pesanan (
    nomer_pesanan VARCHAR(20) PRIMARY KEY,
    invoice VARCHAR(30) UNIQUE NOT NULL,
    waktu TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_kasir VARCHAR(10),
    nama_pembeli VARCHAR(100),
    nim VARCHAR(20),
    no_telp VARCHAR(15),
    alamat TEXT,
    catatan_pemesanan TEXT,
    bukti_pembayaran TEXT,
    status_pesanan VARCHAR(20) DEFAULT 'Proses', -- Bisa 'Proses' atau 'Selesai'
    total_bayar DECIMAL(10, 2) DEFAULT 0,
    FOREIGN KEY (id_kasir) REFERENCES kasir(id_kasir)
);

-- Tabel Detail Pesanan (Keranjang per Transaksi)
CREATE TABLE detail_pesanan (
    id_detail SERIAL PRIMARY KEY, -- Menggunakan Auto Increment
    nomer_pesanan VARCHAR(20),
    id_produk VARCHAR(10),
    qty INT NOT NULL,
    harga_satuan DECIMAL(10, 2) NOT NULL, -- Harga jual saat transaksi terjadi
    harga_modal_satuan DECIMAL(10, 2) NOT NULL, -- Disimpan untuk hitung laba kotor nanti
    subtotal_harga DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (nomer_pesanan) REFERENCES pesanan(nomer_pesanan),
    FOREIGN KEY (id_produk) REFERENCES produk(id_produk)
);
