E. Halaman History
Menampilkan daftar pesanan hari ini yang sudah diselesaikan ("Selesai").

SQL
SELECT p.id_produk, p.nama_produk, ps.nomer_pesanan, ps.invoice, ps.waktu, 
k.nama AS kasir, ps.total_bayar
FROM pesanan ps
JOIN kasir k ON ps.id_kasir = k.id_kasir
JOIN detail_pesanan dp ON ps.nomer_pesanan = dp.nomer_pesanan
JOIN produk p ON dp.id_produk = p.id_produk
WHERE DATE(ps.waktu) = CURRENT_DATE 
AND ps.status_pesanan = 'Selesai';

F. Halaman Settlement (Tutup Kasir)
Menarik data penjualan untuk hari ini saja guna memverifikasi uang fisik dan pendapatan.

SQL
-- 1. Menampilkan detail menu terjual hari ini beserta qty dan subtotal
SELECT p.nama_produk, SUM(dp.qty) as total_qty, SUM(dp.subtotal_harga) as total_penjualan_produk
FROM detail_pesanan dp
JOIN pesanan ps ON dp.nomer_pesanan = ps.nomer_pesanan
JOIN produk p ON dp.id_produk = p.id_produk
WHERE DATE(ps.waktu) = CURRENT_DATE AND ps.status_pesanan = 'Selesai'
GROUP BY p.nama_produk;

-- 2. Menampilkan ringkasan keuangan hari ini
SELECT 
    COUNT(nomer_pesanan) AS total_transaksi,
    SUM(total_bayar) AS subtotal_penjualan
    -- Jika ada pajak, Anda bisa kalikan subtotal_penjualan misal dengan 0.11 (11%)
FROM pesanan 
WHERE DATE(waktu) = CURRENT_DATE AND status_pesanan = 'Selesai';

G. Halaman Report (Laporan Penjualan Detail)
Menarik data secara dinamis berdasarkan inputan rentang tanggal (dari tanggal X sampai tanggal Y).

SQL
-- Parameter misal: dari '2024-01-01' sampai '2024-01-31'

-- 1. Ringkasan Laporan
SELECT 
    COUNT(ps.nomer_pesanan) AS total_transaksi,
    SUM(dp.qty) AS produk_terjual,
    SUM(dp.qty * dp.harga_modal_satuan) AS total_modal_hpp,
    SUM(ps.total_bayar) AS total_pendapatan_akhir,
    (SUM(ps.total_bayar) - SUM(dp.qty * dp.harga_modal_satuan)) AS laba_kotor
FROM pesanan ps
JOIN detail_pesanan dp ON ps.nomer_pesanan = dp.nomer_pesanan
WHERE DATE(ps.waktu) BETWEEN '2024-01-01' AND '2024-01-31'
AND ps.status_pesanan = 'Selesai';

-- 2. List Tabel Daftar Seluruh Transaksi
SELECT ps.invoice, ps.waktu, k.nama AS kasir, ps.total_bayar
FROM pesanan ps
JOIN kasir k ON ps.id_kasir = k.id_kasir
WHERE DATE(ps.waktu) BETWEEN '2024-01-01' AND '2024-01-31'
AND ps.status_pesanan = 'Selesai'
ORDER BY ps.waktu DESC;
